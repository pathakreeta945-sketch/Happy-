    <script>
        // --- Navigation System & Engine State ---
        let currentStep = 1;
        const totalSteps = 5;

        function navigateToStep(targetStep) {
            if (targetStep < 1 || targetStep > totalSteps) return;

            const currentCard = document.getElementById(`step-${currentStep}`);
            const nextCard = document.getElementById(`step-${targetStep}`);
            
            // Deactivate interactives immediately during transformation
            currentCard.classList.remove('active');
            
            // GSAP Sequence Matrix
            const tl = gsap.timeline({
                onComplete: () => {
                    nextCard.classList.add('active');
                    currentStep = targetStep;
                }
            });

            // Transition running profile outbound
            tl.to(currentCard, {
                opacity: 0,
                y: -30,
                scale: 0.95,
                duration: 0.4,
                ease: "power2.in"
            });

            // Prepare next element properties
            gsap.set(nextCard, { y: 40, scale: 0.93, opacity: 0 });

            // Transition running inbound
            tl.to(nextCard, {
                opacity: 1,
                y: 0,
                scale: 1,
                duration: 0.5,
                ease: "power3.out"
            }, "-=0.15");

            // Global progress system update execution
            const progressBar = document.getElementById('progress-bar');
            progressBar.style.width = `${(targetStep / totalSteps) * 100}%`;
        }

        // --- Step 4: Polaroid 3D Parallax & Tracking System ---
        const polaroidElement = document.getElementById('polaroid-element');
        
        if (polaroidElement) {
            const handleMove = (e) => {
                const rect = polaroidElement.getBoundingClientRect();
                // Check if touch or mouse event
                const clientX = e.touches ? e.touches[0].clientX : e.clientX;
                const clientY = e.touches ? e.touches[0].clientY : e.clientY;
                
                const x = clientX - rect.left - rect.width / 2;
                const y = clientY - rect.top - rect.height / 2;
                
                // Map bounding coordinates to rotation constraints
                const rotX = -y / 6; 
                const rotY = x / 6;
                
                gsap.to(polaroidElement, {
                    rotateX: rotX,
                    rotateY: rotY,
                    rotateZ: rotY * 0.1,
                    scale: 1.03,
                    duration: 0.2,
                    ease: "power1.out"
                });
            };

            const handleLeave = () => {
                gsap.to(polaroidElement, {
                    rotateX: 10,
                    rotateY: -5,
                    rotateZ: -2,
                    scale: 1,
                    duration: 0.6,
                    ease: "elastic.out(1, 0.5)"
                });
            };

            polaroidElement.addEventListener('mousemove', handleMove);
            polaroidElement.addEventListener('mouseleave', handleLeave);
            polaroidElement.addEventListener('touchmove', handleMove, { passive: true });
            polaroidElement.addEventListener('touchend', handleLeave);
        }

        // --- Layer 2: Three.js 3D Floating Heart Engine Ecosystem ---
        let scene, camera, renderer, hearts = [];
        let executionFinaleTriggered = false;
        const container = document.getElementById('webgl-container');

        function initThreeEngine() {
            scene = new THREE.Scene();
            
            camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.z = 30;

            renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            container.appendChild(renderer.domElement);

            // Light Matrix Initialization
            const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);
            scene.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffb6c1, 1.2);
            directionalLight.position.set(10, 20, 15);
            scene.add(directionalLight);

            const pointLight = new THREE.PointLight(0xf43f5e, 1.5, 50);
            pointLight.position.set(-10, -10, 10);
            scene.add(pointLight);

            // Construct heart parametric profile structural blueprint paths
            const heartShape = new THREE.Shape();
            heartShape.moveTo(25, 25);
            heartShape.bezierCurveTo(25, 25, 20, 0, 0, 0);
            heartShape.bezierCurveTo(-30, 0, -30, 35, -30, 35);
            heartShape.bezierCurveTo(-30, 55, -10, 77, 25, 95);
            heartShape.bezierCurveTo(70, 77, 90, 55, 90, 35);
            heartShape.bezierCurveTo(90, 35, 90, 0, 60, 0);
            heartShape.bezierCurveTo(40, 0, 25, 25, 25, 25);

            // Structural parameters extrusion mapping profile transformation
            const extrudeSettings = {
                depth: 12,
                bevelEnabled: true,
                bevelSegments: 5,
                steps: 2,
                bevelSize: 4,
                bevelThickness: 4
            };

            const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
            geometry.center(); // Center local coordinate system boundary matrix

            // Instantiation Parameter Engine Arrays
            const totalHearts = 25;
            const colors = [0xf43f5e, 0xec4899, 0xf472b6, 0xff758f, 0xff85a1];

            for(let i=0; i<totalHearts; i++) {
                const colorSelected = colors[Math.floor(Math.random() * colors.length)];
                
                // Advanced Shader Lighting Standard Material Matrix Configuration
                const material = new THREE.MeshStandardMaterial({
                    color: colorSelected,
                    roughness: 0.15,
                    metalness: 0.1,
                    clearcoat: 1.0,
                    clearcoatRoughness: 0.1
                });

                const mesh = new THREE.Mesh(geometry, material);
                
                // Physical transform tracking parameters allocations
                const scaleFactor = 0.015 + Math.random() * 0.02;
                mesh.scale.set(scaleFactor, scaleFactor, scaleFactor);
                
                mesh.position.set(
                    (Math.random() - 0.5) * 45,
                    (Math.random() - 0.5) * 40 - 5,
                    (Math.random() - 0.5) * 20 - 5
                );

                mesh.rotation.set(
                    Math.random() * Math.PI,
                    Math.random() * Math.PI,
                    Math.random() * Math.PI
                );

                // Tracking dynamic mathematical transforms logic definitions
                mesh.userData = {
                    speedY: 0.02 + Math.random() * 0.04,
                    speedRotX: 0.005 + Math.random() * 0.01,
                    speedRotY: 0.005 + Math.random() * 0.01,
                    waveFrequency: 0.5 + Math.random() * 1.5,
                    waveAmplitude: 0.02 + Math.random() * 0.05,
                    seed: Math.random() * 100,
                    initialX: mesh.position.x
                };

                hearts.push(mesh);
                scene.add(mesh);
            }

            animateThreeEngineLoop();
        }

        function animateThreeEngineLoop() {
            requestAnimationFrame(animateThreeEngineLoop);
            const time = window.performance.now() * 0.001;

            hearts.forEach((heart) => {
                if (!executionFinaleTriggered) {
                    // Standard floating runtime behavior
                    heart.position.y += heart.userData.speedY;
                    heart.position.x = heart.userData.initialX + Math.sin(time * heart.userData.waveFrequency + heart.userData.seed) * heart.userData.waveAmplitude * 10;
                    
                    heart.rotation.x += heart.userData.speedRotX;
                    heart.rotation.y += heart.userData.speedRotY;

                    // Boundaries recycling tracking loop logic execution
                    if (heart.position.y > 25) {
                        heart.position.y = -25;
                        heart.position.x = (Math.random() - 0.5) * 45;
                        heart.userData.initialX = heart.position.x;
                    }
                } else {
                    // Explosion Swirl Acceleration Vectors Outbound Triggered
                    heart.position.y += heart.userData.speedY * 5;
                    heart.position.x += (heart.position.x - scene.position.x) * 0.04; 
                    heart.rotation.x += heart.userData.speedRotX * 8;
                    heart.rotation.y += heart.userData.speedRotY * 8;
                    
                    // Controlled gradual scaling degradation down to zero vector profile
                    if(heart.scale.x > 0.0001) {
                        heart.scale.subScalar(0.0002);
                    }
                }
            });

            renderer.render(scene, camera);
        }

        // Window resize responsive adjustment interface execution handler
        window.addEventListener('resize', () => {
            if(!camera || !renderer) return;
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        // Initialize engine framework on system launch phase load window execution
        window.addEventListener('DOMContentLoaded', () => {
            initThreeEngine();
            // Injects absolute 100% complete state inside fixed progress display baseline
            document.getElementById('progress-bar').style.width = `${(currentStep / totalSteps) * 100}%`;
        });

        // --- Grand Finale Climax Sequence ---
        function triggerGrandFinaleSequence() {
            const celebrateBtn = document.getElementById('celebrate-btn');
            const finalMessageContainer = document.getElementById('final-message-container');
            
            // Step 1: Disable/Fade Button Execution
            celebrateBtn.disabled = true;
            gsap.to(celebrateBtn, {
                opacity: 0,
                y: 15,
                scale: 0.9,
                duration: 0.4,
                ease: "power2.inOut",
                onComplete: () => {
                    celebrateBtn.style.display = 'none';
                }
            });

            // Step 2: Inbound final wish message configuration rendering structural pipeline animations
            finalMessageContainer.style.height = "auto";
            gsap.to(finalMessageContainer, {
                opacity: 1,
                y: -10,
                duration: 0.8,
                delay: 0.2,
                ease: "power3.out"
            });

            // Step 3: Trigger 3D Hearts Swirl Outward Logic State Vector Change
            executionFinaleTriggered = true;

            // Step 4: Multi-Layered Confetti Impact Matrix Calculations
            // Burst Phase A: Bottom Corner Side Cannons
            const end = Date.now() + (5 * 1000); // 5 Seconds execution profile baseline metrics
            
            confetti({
                particleCount: 80,
                spread: 60,
                origin: { x: 0, y: 0.9 }
            });
            confetti({
                particleCount: 80,
                spread: 60,
                origin: { x: 1, y: 0.9 }
            });

            // Continuous mixed emoji structural rainfall loop system framework execution logic
            (function frameConfettiShowerEngineLoop() {
                if (Date.now() > end) return;

                // Standard Colorized Foil Streamers
                confetti({
                    particleCount: 3,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0, y: 0.8 }
                });
                confetti({
                    particleCount: 3,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1, y: 0.8 }
                });

                // Heart and Sparkle Scaled Particle Drops Matrix Elements Injection
                confetti({
                    particleCount: 2,
                    spread: 120,
                    origin: { x: Math.random(), y: Math.random() * 0.3 },
                    colors: ['#f43f5e', '#ec4899', '#ffb6c1'],
                    scalar: 1.2
                });

                requestAnimationFrame(frameConfettiShowerEngineLoop);
            }());
        }
    </script>
